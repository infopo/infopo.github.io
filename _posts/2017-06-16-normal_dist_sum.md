

```python
mu = 175
var = 4
height = np.random.normal(mu, var, 20)

def norm_dist():
    return [(1 / (np.sqrt(2*np.pi*var))) * np.exp(-((x-mu)**2)/(2*var)) for x in height]

dfh = pd.DataFrame({'height':height, 'prob':norm_dist()})
dfh.head()
```
